\section{Easy}
\begin{problem}
    Evaluate the following 
    \[
      \int_0^1 x^2 \sqrt{1-x^2} \dd x 
    \]
\end{problem}
Consider the u sub:
\begin{align*}
  x&=\sin u \\
  \dd x&= \cos u \dd u \\
\end{align*}
When $x=0,u=0$ and $x=1,u=\frac{\pi}{2}$
\begin{align*}
  \int_0^1 x^2 \sqrt{1-x^2} \dd x &= \int_0^{\pi/2} \sin^2u\cos^2u \dd u \\
  &= \int_0^{\pi/2} \frac{1}{4}(1-\cos 2u)(1+\cos 2u) \dd u \\
  &= \frac{1}{4} \int_0^{\pi/2} 1-\cos^2 2u \dd u \\
  &= \frac{1}{4} \int_0^{\pi/2} 1-\frac{1}{2}(1+\cos4u) \dd u \\
  &= \frac{1}{8} \int_0^{\pi/2} 1-\cos 4u \dd u \\
  &= \frac{1}{8} \Big[ u-\frac{\sin 4u}{4} \Big]_{0}^{\pi / 2} \\
  &= \frac{\pi}{16}
\end{align*}
