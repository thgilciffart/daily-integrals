\section{Easy}
\begin{problem}
    Evaluate the following \[
    \int^{\pi/4}_0\frac{\sin x}{\sqrt{1+\sin^2x}}\dd x
    \]
\end{problem}
Consider the sub $u=\sin x$
\begin{align*}
    u&=\sin x \\
    \dd u &= \cos x \dd x
\end{align*}
And when $x=0,u=0$, and $x=\dfrac{\pi}{4},u=\dfrac{\sqrt{2}}{2}$
\begin{align*}
    \int^{\pi/4}_0\frac{\sin x}{\sqrt{1+\sin^2x}}\dd x &= \int^{\sqrt{2}/2}_0\frac{u}{\sqrt{1+u^2}}\frac{1}{\cos x }\dd u \\
    &=\int^{\sqrt{2}/2}_0\frac{u}{\sqrt{1+u^2}}\frac{1}{\sqrt{1-u^2}}\dd u \\
    &=\int^{\sqrt{2}/2}_0\frac{u}{\sqrt{1-u^4}}\dd u \\
\end{align*}
Consider the sub $u=\sqrt{\sin w}$
\begin{align*}
    u&=\sqrt{\sin w} \\ 
    \dd u &= \frac{\cos w}{2\sqrt{\sin w}} \dd w \\
    \dd u &= \frac{\sqrt{1-\sin^2w}}{2u} \dd w \\
    \dd u &= \frac{\sqrt{1-u^4}}{2u} \dd w \\
\end{align*}
And when $u=0,w=0$, and $u=\dfrac{\sqrt{2}}{2},w=\dfrac{\pi}{6}$. 
\begin{align*}
    \int^{\sqrt{2}/2}_0\frac{u}{\sqrt{1-u^4}}\dd u &= \int^{\pi/6}_0\frac{u}{\sqrt{1-u^4}}\frac{\sqrt{1-u^4}}{2u} \dd w \\
    &= \frac{1}{2}\int^{\pi/6}_0 \dd w \\
    &= \frac{1}{2} \Big[w \Big]^{\pi/6}_0 \\
    &= \frac{\pi}{12}
\end{align*}
\section{Medium}
\begin{problem}
    Evaluate the following 
    \[
    \int_0^\pi\frac{x^2\cos^2x-x\sin x-\cos x-1}{(1+x\sin x)^2}\dd x
    \]
\end{problem}
\begin{align*}
        \int_0^\pi\frac{x^2\cos^2x-x\sin x-\cos x-1}{(1+x\sin x)^2}\dd x &= \int_0^\pi \frac{x^2(1-\sin^2x)-x\sin x-\cos x-1}{1+2x\sin x +x^2\sin^2x} \dd x \\
        &=\int_0^\pi \frac{x^2-x^2\sin^2x-x\sin x-\cos x - 1}{1+2x\sin x + x^2\sin^2 x} \dd x \\
        &=\int_0^\pi\frac{x^2-\cos x+x\sin x-(1+x^2\sin^2 x+2x\sin x)}{1+2x\sin x + x^2\sin^2 x} \dd x \\
        &=\int_0^\pi \frac{x^2+x\sin x -\cos x}{(1+x\sin x)^2}\dd x -\int_0^\pi \dd x \\
        &= \int_0^\pi \frac{x^2+x\sin x -\cos x}{(1+x\sin x)^2}\dd x - \int_0^\pi \dd x \\
\end{align*}
