\section{Easy}
\begin{problem}
    Evaluate the following \[
    \int^{1}_0 \frac{\arctan\sqrt{x}}{\sqrt{x}(1+x)} \dd x
    \]
\end{problem}
\begin{align*}
    u&=\arctan\sqrt{x} \\
    \dd u &= \frac{1}{2\sqrt{x}(1+x)} \dd x \\
    \dd x &= 2\sqrt{x}(1+x) \dd u
\end{align*}
and $x=0,u=0$  $x=1,u=\dfrac{\pi}{4}$
\begin{align*}
     \int^{1}_0 \frac{\arctan\sqrt{x}}{\sqrt{x}(1+x)} \dd x &= \int_0^{\pi/4}\frac{u}{\sqrt{x}(1+x)} 2\sqrt{x}(1+x) \dd u \\
     &=\int_0^{\pi/4}2u \dd u \\
     &=\Big[u^2\Big]_0^{\pi/4} \\
     &=\frac{\pi^2}{16} \\
     &\approx 0.62
\end{align*}

\section{Medium}
\begin{problem}
    Evaluate the following \[
    \int_{1}^{2}\frac{\left[\ln\left(x^{2}+1\right)-2\ln x\right]\sqrt{x^{2}+1}}{x^{4}} \dd x
    \]
\end{problem}
\begin{align*}
    \int_{1}^{2}\frac{\left[\ln\left(x^{2}+1\right)-2\ln x\right]\sqrt{x^{2}+1}}{x^{4}} \dd x &=\int_{1}^{2}\frac{\ln\left(\frac{x^{2}+1}{x^{2}}\right)\sqrt{x^{2}+1}}{x^{4}} \dd x \\
    &= \int_{1}^{2}\frac{\ln\left(1+\frac{1}{x^{2}}\right)\sqrt{x^{2}\left(1+\frac{1}{x^{2}}\right)}}{x^{4}} \dd x \\
    &=\int_{1}^{2}\frac{\ln\left(1+\frac{1}{x^{2}}\right)\sqrt{\left(1+\frac{1}{x^{2}}\right)}}{x^{3}} \dd x
\end{align*}
\begin{align*}
    u&=\sqrt{1++\frac{1}{x^2}} \\
    \dd u &= \frac{-1}{x^3\sqrt{1+\frac{1}{x^2}}} \dd x \\
    \dd x &= -x^3\sqrt{1+\frac{1}{x^2}} \dd u
\end{align*}
$x=1,u=\sqrt{2}$ and $x=2,u=\dfrac{\sqrt{5}}{2}$
\begin{align*}
    \int_{1}^{2}\frac{\ln\left(1+\frac{1}{x^{2}}\right)\sqrt{\left(1+\frac{1}{x^{2}}\right)}}{x^{3}} \dd x &=-\int_{\sqrt{2}}^{\sqrt{5}/2}\frac{\ln (u^2)u}{x^{3}} x^3\sqrt{1+\frac{1}{x^2}} \dd u \\
    &=-\int_{\sqrt{2}}^{\sqrt{5}/2} \ln (u^2)u^2\dd u \\
    &=-2\int_{\sqrt{2}}^{\sqrt{5}/2} \ln (u)u^2\dd u
\end{align*}
Using IBP
\begin{align*}
    -2\int_{\sqrt{2}}^{\sqrt{5}/2} \ln (u)u^2\dd u&=-2 \left[ \frac{u^3}{3}\ln u \right]_{\sqrt{2}}^{\sqrt{5}/2} +2\int_{\sqrt{2}}^{\sqrt{5}/2} \frac{1}{u}\frac{u^3}{3} \dd u \\
    &=-2 \left[ \frac{u^3}{3}\ln u \right]_{\sqrt{2}}^{\sqrt{5}/2} + \frac{2}{3}\int_{\sqrt{2}}^{\sqrt{5}/2} u^2 \dd u \\
    &=-2 \left[ \frac{u^3}{3}\ln u \right]_{\sqrt{2}}^{\sqrt{5}/2} + \frac{2}{3}\left[\frac{u^3}{3}\right]_{\sqrt{2}}^{\sqrt{5}/2} \\
    &=-\frac{2}{3} \left[ u^3\ln u \right]_{\sqrt{2}}^{\sqrt{5}/2} + \frac{2}{9}\left[u^3\right]_{\sqrt{2}}^{\sqrt{5}/2} \\
    &\approx 0.23
\end{align*}
